    tar xzvf Statistics-Descriptive-3.0612.tar.gz
    perl Makefile.PL PREFIX=/work-zfs/mschatz1/mschatz/lib/perl5/site_perl
    make
    make test
    make install
    export PERL5LIB="/work-zfs/mschatz1/mschatz/lib/perl5/site_perl/share/perl5"
